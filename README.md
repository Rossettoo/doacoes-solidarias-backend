CREATE TABLE public.usuarios (
	id int4 GENERATED BY DEFAULT AS IDENTITY( INCREMENT BY 1 MINVALUE 1 MAXVALUE 2147483647 START 1 CACHE 1 NO CYCLE) NOT NULL,

	email varchar(255) NULL,

	"role" varchar(255) NULL,
